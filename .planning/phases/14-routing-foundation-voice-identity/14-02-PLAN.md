# ---- Intent Classification (regex, <0.01ms) ----

# Route patterns — checked in priority order: smithers > deity > jarvis
_SMITHERS_PATTERN = re.compile(
    r'\b(?:connect.*smithers|talk.*smithers|smithers|ask smithers|hey smithers|'
    r'slack|obsidian|vault|my notes|schedule|calendar|meeting|appointment|'
    r'todo|task|send message|post to|remind|to.?do list|tasks|channel)\b',
    re.IGNORECASE
)

_DEITY_NAMES_PATTERN = re.compile(
    r'\b(?:zeus|athena|apollo|ares|hermes|poseidon|aphrodite|artemis|'
    r'hera|hephaestus|dionysus|hades)\b',
    re.IGNORECASE
)

_DEITY_DOMAIN_PATTERN = re.compile(
    r'\b(?:olympus|mythology|myth|oracle|prophecy|divine|sacred|'
    r'goddess|god of|titan|mortal|underworld|mount olympus)\b',
    re.IGNORECASE
)


def classify_intent(text: str, current_voice: str = "jarvis") -> dict:
    """Classify a voice query into a route and voice. Synchronous, <0.01ms.

    Route priority: smithers > deity > jarvis (fallback).
    Returns: {"route": str, "voice": str, "confidence": str}
    """
    # Priority 1: Smithers (task/productivity)
    if _SMITHERS_PATTERN.search(text):
        return {"route": "smithers", "voice": "jarvis", "confidence": "high"}

    # Priority 2: Deity (mythology) — check for specific deity name first
    deity_match = _DEITY_NAMES_PATTERN.search(text)
    if deity_match:
        deity_name = deity_match.group(0).lower()
        # Validate against VOICES dict
        voice = deity_name if deity_name in VOICES else "athena"
        return {"route": "deity", "voice": voice, "confidence": "high"}

    # Priority 2b: Deity domain terms without a specific name
    if _DEITY_DOMAIN_PATTERN.search(text):
        # Use current voice if it's already a deity, otherwise default to athena
        voice = current_voice if current_voice in DEITY_PROMPTS else "athena"
        return {"route": "deity", "voice": voice, "confidence": "medium"}

    # Priority 3: Jarvis fallback
    return {"route": "jarvis", "voice": current_voice if current_voice in VOICES else "jarvis", "confidence": "low"}
```

Key design decisions per research:
- **Regex only, NOT LLM** — Haiku measured at 1,667ms, regex at <0.01ms. Per user decision.
- **smithers > deity priority** — "Schedule a call with Zeus" routes to Smithers, not deity.
- **Voice validated against VOICES.keys()** — prevents KeyError crashes from unknown voice names.
- **current_voice preserved for jarvis fallback** — if user is already talking to a deity and says something general, stays on that voice.
- **Separate deity name vs domain patterns** — "Tell me about Zeus" gets Zeus voice, "Tell me a myth" gets current deity or Athena default.

async def process_query(websocket, text: str, voice: str, model: str = None, tier: str = None):
    """Process a text query: classify → route → TTS → send audio."""
    session_id = str(id(websocket))

    # ---- Intent Classification (synchronous, <0.01ms) ----
    intent = classify_intent(text, voice)
    route = intent["route"]
    classified_voice = intent["voice"]

    print(f"  [CLASSIFY] route={route}, voice={classified_voice}, confidence={intent['confidence']}, text=\"{text[:60]}\"")

    # ---- Voice-Role Binding (lock voice before any async work) ----
    if classified_voice != voice:
        voice = classified_voice
        # Validate against VOICES dict (safety net)
        if voice not in VOICES:
            print(f"  [VOICE] Unknown voice '{voice}', falling back to jarvis")
            voice = "jarvis"
        await websocket.send(json.dumps({"type": "voice_changed", "voice": voice}))

    # ---- Send thinking state ----
    await websocket.send(json.dumps({"type": "thinking", "voice": voice}))

    # ---- Route dispatch ----
    if route == "smithers":
        # Strip smithers trigger phrases from query for cleaner Smithers input
        clean_text = re.sub(
            r'\b(?:connect.*smithers|talk.*smithers|smithers|ask smithers|hey smithers)\b',
            '', text, flags=re.IGNORECASE
        ).strip() or text
        result = await route_smithers(clean_text, voice, None, None, session_id=session_id)
    elif route == "deity":
        result = await route_deity(text, voice, model, tier, session_id=session_id)
    else:
        # jarvis fallback — use route_query which checks DEITY_PROMPTS for current voice
        result = await route_query(text, voice, model, tier, session_id=session_id)

    reply = result["text"]
    model_used = result["model_used"]

    # ---- Single unified response + TTS pipeline ----
    await websocket.send(json.dumps({
        "type": "response", "text": reply, "voice": voice, "model": model_used,
    }))

    # Trigger magic sparkle for dramatic deity responses
    dramatic_keywords = re.compile(r'\b(?:behold|listen|hear me|mortal|prophecy|fate|destiny|thunder|unleash|fury|sacred|divine|eternal|curse|blessing|oracle)\b', re.IGNORECASE)
    if voice in DEITY_WIREFRAMES and dramatic_keywords.search(reply):
        await websocket.send(json.dumps({"type": "magic_sparkle"}))
        print(f"  [SPARKLE] Dramatic response detected")

    wireframe = pick_wireframe(reply, voice)
    if wireframe:
        await websocket.send(json.dumps({"type": "wireframe", "model": wireframe}))

    voice_id = VOICES.get(voice, VOICES["jarvis"])["id"]
    await websocket.send(json.dumps({"type": "speaking"}))
    audio = await get_tts_audio(reply, voice_id)
    if audio:
        for i in range(0, len(audio), 4096):
            await websocket.send(audio[i:i + 4096])
            await asyncio.sleep(0.01)
    await websocket.send(json.dumps({"type": "done", "has_follow_up": has_follow_up(reply)}))
    print(f"[DONE]\n")
```
