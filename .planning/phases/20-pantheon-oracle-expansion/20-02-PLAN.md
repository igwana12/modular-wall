      ```python
      """Ingest orphic chunks into orphic_wisdom ChromaDB collection."""
      import json
      from pathlib import Path
      import chromadb
      from chromadb.utils.embedding_functions import SentenceTransformerEmbeddingFunction

      CHROMA_PATH = Path.home() / ".jarvis" / "myth-vectors"
      CHUNKS_DIR = Path("/tmp/orphic_chunks")
      COLLECTION_NAME = "orphic_wisdom"
      BATCH_SIZE = 50

      ef = SentenceTransformerEmbeddingFunction(
          model_name="all-mpnet-base-v2",
          normalize_embeddings=True,
      )
      client = chromadb.PersistentClient(path=str(CHROMA_PATH))
      collection = client.get_or_create_collection(
          name=COLLECTION_NAME,
          embedding_function=ef,
          metadata={"hnsw:space": "cosine"},
      )

      chunk_files = sorted(CHUNKS_DIR.glob("*.json"))
      print(f"Found {len(chunk_files)} chunk files")

      ids, docs, metas = [], [], []
      for i, cf in enumerate(chunk_files):
          data = json.loads(cf.read_text())
          chunk_id = data.get("id", cf.stem)
          text = data.get("text", data.get("content", ""))
          meta = {k: v for k, v in data.items() if k not in ("text", "content", "id") and isinstance(v, (str, int, float, bool))}
          meta["source"] = "orphic"
          ids.append(str(chunk_id))
          docs.append(text)
          metas.append(meta)
          if len(ids) >= BATCH_SIZE:
              collection.upsert(ids=ids, documents=docs, metadatas=metas)
              print(f"Upserted batch up to chunk {i+1}")
              ids, docs, metas = [], [], []

      if ids:
          collection.upsert(ids=ids, documents=docs, metadatas=metas)
          print(f"Upserted final batch")

      print(f"Total in collection: {collection.count()}")
      assert collection.count() >= 370, f"Expected ~376, got {collection.count()}"
      print("INGEST COMPLETE")
      ```

      IMPORTANT: chunk files may be .json with varying schemas. If chunks are not
      JSON (e.g., plain text), adapt the read logic: text = cf.read_text(), 
      chunk_id = cf.stem, meta = {"source": "orphic"}.
