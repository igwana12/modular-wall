**Create plotly-panel.js** — a new panel module following the same interface as chart-panel (ECharts):

```javascript
class PlotlyPanel {
    constructor() {
        this.container = document.createElement('div');
        this.container.className = 'viz-panel plotly-panel';
        this.container.style.cssText = 'width: 100%; height: 100%;';
        this._plotDiv = document.createElement('div');
        this._plotDiv.style.cssText = 'width: 100%; height: 100%;';
        this.container.appendChild(this._plotDiv);
    }

    get isVisible() { return this.container.classList.contains('visible'); }

    handleMessage(msg) {
        this.container.classList.add('visible');
        const chartType = msg.chart_type || 'surface';
        const plotData = this._buildPlotlyData(chartType, msg.data, msg.title);
        const layout = this._buildLayout(chartType, msg.title);

        if (this._plotDiv._plotlyInitialized) {
            // Morph: update data in place with animation
            Plotly.animate(this._plotDiv, { data: plotData }, {
                transition: { duration: 500, easing: 'cubic-in-out' },
                frame: { duration: 500 }
            });
        } else {
            Plotly.newPlot(this._plotDiv, plotData, layout, { responsive: true });
            this._plotDiv._plotlyInitialized = true;
        }
    }

    _buildPlotlyData(chartType, data, title) {
        if (chartType === 'surface') {
            return [{ type: 'surface', z: data.z || data.values, colorscale: 'Viridis' }];
        }
        if (chartType === 'scatter3d') {
            return [{ type: 'scatter3d', mode: 'markers',
                x: data.x, y: data.y, z: data.z,
                marker: { size: 4, color: data.z, colorscale: 'Portland' }
            }];
        }
        if (chartType === 'line3d') {
            return [{ type: 'scatter3d', mode: 'lines',
                x: data.x, y: data.y, z: data.z,
                line: { width: 3, color: data.color || '#00d4ff' }
            }];
        }
        return [{ type: 'surface', z: data.z || [[1,2],[3,4]] }];
    }

    _buildLayout(chartType, title) {
        return {
            title: { text: title || '', font: { color: '#ccc', size: 14 } },
            paper_bgcolor: 'rgba(0,0,0,0)',
            plot_bgcolor: 'rgba(0,0,0,0)',
            scene: {
                xaxis: { gridcolor: '#333', zerolinecolor: '#555' },
                yaxis: { gridcolor: '#333', zerolinecolor: '#555' },
                zaxis: { gridcolor: '#333', zerolinecolor: '#555' },
                bgcolor: 'rgba(0,0,0,0)',
            },
            margin: { l: 0, r: 0, t: 40, b: 0 },
            font: { color: '#aaa' },
        };
    }

    hide() {
        this.container.classList.remove('visible');
    }

    destroy() {
        if (this._plotDiv._plotlyInitialized) {
            Plotly.purge(this._plotDiv);
            this._plotDiv._plotlyInitialized = false;
        }
    }
}
```
