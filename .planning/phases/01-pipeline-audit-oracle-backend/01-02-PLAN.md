    **tts.py -- ElevenLabs WebSocket streaming TTS (per D-11)**
    ```python
    import websockets
    import json
    import base64
    import os

    ELEVENLABS_API_KEY = os.getenv("ELEVENLABS_API_KEY")

    async def tts_stream(voice_id: str, text: str):
        """Stream TTS audio for a sentence via ElevenLabs WebSocket.
        Yields base64-encoded MP3 audio chunks.
        Uses Flash v2.5 model for 75ms latency (per D-11)."""
        uri = f"wss://api.elevenlabs.io/v1/text-to-speech/{voice_id}/stream-input"
        params = "?model_id=eleven_flash_v2_5&output_format=mp3_44100_128"

        async with websockets.connect(uri + params) as ws:
            # Initialize
            await ws.send(json.dumps({
                "text": " ",
                "xi_api_key": ELEVENLABS_API_KEY,
                "voice_settings": {"stability": 0.5, "similarity_boost": 0.8},
            }))
            # Send text
            await ws.send(json.dumps({"text": text}))
            # Signal end
            await ws.send(json.dumps({"text": ""}))
            # Receive audio chunks
            async for message in ws:
                data = json.loads(message)
                if data.get("audio"):
                    yield data["audio"]  # already base64 from ElevenLabs
                if data.get("isFinal"):
                    break
    ```

    **pipeline.py -- Core reading pipeline orchestration**
    The pipeline has two modes:
    1. **Streaming mode** (primary): yields token-by-token from Claude for progressive SSE delivery
    2. **Full mode** (fallback): returns complete text if streaming fails

    For streaming, call Claude directly via the `anthropic` SDK (NOT LLM Router, which does not support streaming). The LLM Router can be called as fallback for non-streaming.

    ```python
    import anthropic
    import httpx

    async def execute_reading_stream(deity_config: dict, intent: str, rag_context: str):
        """Stream oracle reading tokens from Claude.
        Yields string tokens as they arrive.
        Uses deity system_prompt with RAG context injected."""

        system_prompt = deity_config["system_prompt"]
        # Inject RAG context into system prompt
        full_system = f"{system_prompt}\n\n## Relevant Mythology Context\n{rag_context}"

        user_message = f"The querent seeks guidance on: {intent}\n\nDeliver an oracle reading as {deity_config['name']}. Speak in character. Apply the McKee narrative arc in your reading. Be specific, personal, and transformative. 150-300 words."

        client = anthropic.AsyncAnthropic()  # Uses ANTHROPIC_API_KEY env var
        async with client.messages.stream(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            system=full_system,
            messages=[{"role": "user", "content": user_message}],
        ) as stream:
            async for text in stream.text_stream:
                yield text
    ```

    **streaming.py -- SSE stream generation with sentence detection (per D-09, D-10)**
    ```python
    import re
    import json
    import asyncio
    from datetime import datetime

    SENTENCE_END = re.compile(r'[.!?]["\')\]]?\s+|[.!?]["\')\]]?$')

    async def stream_reading(deity_config: dict, intent: str):
        """Generator yielding SSE events for an oracle reading.
        Interleaves text tokens and audio chunks progressively.
        Per D-09: progressive delivery, target <1s to first text.
        Per D-10: NO batch-then-stream."""

        # 1. Immediately yield deity metadata
        yield {
            "event": "deity",
            "data": json.dumps({
                "name": deity_config["name"],
                "title": deity_config["title"],
                "image": None,  # Content DB image ref
            })
        }

        # 2. Get RAG context (fast, <500ms)
        from rag import get_reading_context
        rag_context = await get_reading_context(deity_config["id"], intent)

        # 3. Stream LLM tokens with sentence detection
        from pipeline import execute_reading_stream
        sentence_buffer = ""
        full_text = ""

        async for token in execute_reading_stream(deity_config, intent, rag_context):
            full_text += token
            sentence_buffer += token

            # Yield text token immediately (progressive)
            yield {"event": "text", "data": json.dumps({"chunk": token})}

            # Check for sentence boundary
            if SENTENCE_END.search(sentence_buffer):
                sentence = sentence_buffer.strip()
                sentence_buffer = ""

                # Stream TTS for completed sentence
                if deity_config.get("voice_id"):
                    from tts import tts_stream
                    try:
                        async for audio_chunk in tts_stream(deity_config["voice_id"], sentence):
                            yield {"event": "audio", "data": json.dumps({"chunk": audio_chunk})}
                    except Exception as e:
                        # TTS failure is non-fatal -- text reading continues
                        yield {"event": "tts_error", "data": json.dumps({"error": str(e)})}

        # 4. Flush remaining buffer
        if sentence_buffer.strip():
            if deity_config.get("voice_id"):
                from tts import tts_stream
                try:
                    async for audio_chunk in tts_stream(deity_config["voice_id"], sentence_buffer.strip()):
                        yield {"event": "audio", "data": json.dumps({"chunk": audio_chunk})}
                except Exception:
                    pass

        # 5. Signal completion
        yield {"event": "done", "data": json.dumps({"total_chars": len(full_text)})}
    ```
