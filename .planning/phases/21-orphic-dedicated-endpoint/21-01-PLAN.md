    **Theme gate — exact logic:**
    ```python
    def check_theme_gate(question: str, extra_keywords: list[str] | None, force: bool) -> set[str]:
        """Return the set of matched Orphic keywords. Raises HTTPException(422) if <2 matches and force is False.

        Tokenization: lowercase + whitespace split of question + extra_keywords. Only single-word keywords
        from ORPHIC_THEME_KEYWORDS can match (multi-word entries like "gold tablet" never match — see RESEARCH Pitfall 4).
        """
        words = set(question.lower().split())
        if extra_keywords:
            for kw in extra_keywords:
                words.update(kw.lower().split())
        matches = ORPHIC_THEME_KEYWORDS & words
        if len(matches) < 2 and not force:
            raise HTTPException(
                status_code=422,
                detail=(
                    f"Orphic theme gate: need ≥2 keyword matches from the Orphic corpus, "
                    f"found {len(matches)}. Pass force=true to bypass the gate, or add "
                    f"keywords like 'orpheus', 'dionysus', 'death', 'rebirth', 'underworld', 'soul'."
                ),
            )
        return matches
    ```

    **SOUL loader with path-traversal guard:**
    ```python
    def load_soul_file(deity: str | None) -> tuple[str, str]:
        """Load the SOUL persona for the given deity name, with path-traversal guard.

        Returns (persona_text, display_deity_name).
        - If deity is None/empty: returns (GODDESS_FALLBACK_PERSONA, "Goddess").
        - If the resolved path escapes ORPHIC_SOULS_DIR: raises HTTPException(400).
        - If the resolved path is inside ORPHIC_SOULS_DIR but the file does not exist:
              returns (GODDESS_FALLBACK_PERSONA, "Goddess") — silent fallback per CONTEXT.md.
        - Otherwise: returns (file_content, deity.capitalize()).

        Path-traversal guard: resolve the candidate path, then verify it is_relative_to
        ORPHIC_SOULS_DIR. This blocks "../../etc/passwd" and absolute-path injection.
        Deity names are further restricted to [a-z0-9_-]+ — any other character triggers 400.
        """
        if not deity or not deity.strip():
            return GODDESS_FALLBACK_PERSONA, "Goddess"

        deity_clean = deity.strip().lower()

        # Character whitelist — deity names are lowercase ASCII letters only (matches the 20 SOUL files)
        import re as _re
        if not _re.fullmatch(r"[a-z][a-z0-9_-]*", deity_clean):
            raise HTTPException(
                status_code=400,
                detail=f"Invalid deity name: must match [a-z][a-z0-9_-]* (got {deity!r})",
            )

        candidate = (ORPHIC_SOULS_DIR / f"SOUL-{deity_clean}.md").resolve()
        try:
            candidate.relative_to(ORPHIC_SOULS_DIR)
        except ValueError:
            raise HTTPException(
                status_code=400,
                detail=f"Invalid deity name: path traversal rejected (got {deity!r})",
            )

        if not candidate.exists():
            logger.warning(f"[Orphic] SOUL file not found for deity={deity_clean!r}, falling back to Goddess")
            return GODDESS_FALLBACK_PERSONA, "Goddess"

        display_name = deity_clean.capitalize()
        return candidate.read_text(encoding="utf-8"), display_name
    ```

        # 3. RAG retrieval (Orphic-only)
        auto_keywords = [w for w in req.question.split() if len(w) > 3][:10]
        if req.keywords:
            auto_keywords = list(req.keywords) + auto_keywords
        if display_deity != "Goddess":
            auto_keywords.insert(0, display_deity.lower())

        wisdom = retrieve_oracle_wisdom(
            hexagram_name="",
            archetype_names=[display_deity] if display_deity != "Goddess" else [],
            keywords=auto_keywords,
            monomyth_stage="",
            question=req.question,
            orphic_only=True,
        )
        wisdom_text = format_wisdom_for_llm(wisdom, "")[:2500]

        # 4. Claude synthesis
        user_message = (
            f"Deity: {display_deity}\n\n"
            f"SOUL persona context:\n{soul_text}\n\n"
            f"Seeker's question: {req.question}\n\n"
            f"Orphic corpus fragments:\n{wisdom_text or 'No corpus fragments retrieved — speak from the deep tradition.'}"
        )

        client = anthropic.AsyncAnthropic()
        response = await client.messages.create(
            model=CLAUDE_MODEL,
            max_tokens=CLAUDE_MAX_TOKENS,
            system=ORPHIC_MCKEE_SYSTEM_PROMPT,
            messages=[{"role": "user", "content": user_message}],
        )
        raw = response.content[0].text.strip()
        if raw.startswith("```"):
            # Strip markdown fence: take content between first and last fence
            raw = raw.split("\n", 1)[1].rsplit("```", 1)[0].strip()

        try:
            parsed = json.loads(raw)
        except json.JSONDecodeError as e:
            logger.error(f"[Orphic] Claude returned unparseable JSON: {raw[:200]!r}")
            raise HTTPException(status_code=502, detail=f"LLM returned unparseable JSON: {e}")

        required_keys = ("inciting_incident", "crisis", "climax", "resolution")
        missing = [k for k in required_keys if not parsed.get(k)]
        if missing:
            logger.error(f"[Orphic] Claude response missing required keys: {missing}")
            raise HTTPException(status_code=502, detail=f"LLM response missing keys: {missing}")

        return OrphicReadingResponse(
            inciting_incident=parsed["inciting_incident"],
            crisis=parsed["crisis"],
            climax=parsed["climax"],
            resolution=parsed["resolution"],
            deity=display_deity,
        )

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP client → FastAPI route | Untrusted JSON body crosses here. The `deity`, `question`, `keywords`, and `force` fields are all client-controlled. |
| FastAPI route → filesystem | The `deity` field is used to construct a path into `/Users/claw2501/niko-obsidian-vault/PROJECTS/The Orb/personas/`. Path traversal is a live risk. |
| FastAPI route → Anthropic API | `question`, `deity`-loaded SOUL text, and RAG corpus fragments are interpolated into a prompt sent to external Claude API. Prompt-injection via the question field is a possibility. |
| FastAPI route → ChromaDB (local) | Keywords derived from user question are passed to `retrieve_oracle_wisdom`. ChromaDB query strings are safe (no SQL-like injection surface), but unbounded keyword lists could cause slow queries. |
