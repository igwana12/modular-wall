```bash
#!/bin/bash
# orb-backend health poll — every 5 minutes
# Alerts to #the-orb via Smithers ONLY when service is DOWN (per D-11)
export PATH=/opt/homebrew/bin:$PATH

STATE_FILE="/tmp/orb-health-state"
LOG_FILE="/Users/claw2501/.openclaw/workspace/logs/orb-health-poll.log"

# Check if orb-backend responds
if curl -s --max-time 5 "http://localhost:8300" > /dev/null 2>&1; then
    # Service is UP — clear state if it was previously down
    if [ -f "$STATE_FILE" ]; then
        rm "$STATE_FILE"
        # Recovery alert
        curl -s -X POST http://localhost:8200/execute/v2 \
            -H "Content-Type: application/json" \
            -d '{"task":"Post to #the-orb: orb-backend (:8300) has RECOVERED and is responding again."}' \
            >> "$LOG_FILE" 2>&1
    fi
else
    # Service is DOWN
    if [ ! -f "$STATE_FILE" ]; then
        # First detection — alert
        echo "$(date)" > "$STATE_FILE"
        curl -s -X POST http://localhost:8200/execute/v2 \
            -H "Content-Type: application/json" \
            -d '{"task":"Post ALERT to #the-orb: orb-backend (:8300) is UNREACHABLE. Detected at '"$(date)"'. Investigate immediately."}' \
            >> "$LOG_FILE" 2>&1
    fi
    # If state file exists, we already alerted — don't spam (per D-11)
fi
```

Key design decisions:
- Uses `/tmp/orb-health-state` to track state — only alerts on state CHANGE (up->down, down->up)
- No degraded performance alerts (per D-11 — down-only)
- Sends recovery alert when service comes back (useful for knowing issue resolved)
- 5-second curl timeout so script completes well within 5-minute interval

```bash
#!/bin/bash
# orb-backend.log error scanner — every 30 minutes
# Alerts to #the-orb via Smithers when errors found
export PATH=/opt/homebrew/bin:$PATH

LOG_FILE="/Users/claw2501/services/orb-backend/orb-backend.log"
SCAN_LOG="/Users/claw2501/.openclaw/workspace/logs/log-scanner.log"
LAST_SCAN_FILE="/tmp/log-scanner-last-pos"

if [ ! -f "$LOG_FILE" ]; then
    exit 0
fi

# Get last scan position (byte offset)
LAST_POS=0
if [ -f "$LAST_SCAN_FILE" ]; then
    LAST_POS=$(cat "$LAST_SCAN_FILE")
fi

# Get current file size
CURRENT_SIZE=$(stat -f%z "$LOG_FILE" 2>/dev/null || stat --format=%s "$LOG_FILE" 2>/dev/null)

# If file was truncated/rotated, reset position
if [ "$CURRENT_SIZE" -lt "$LAST_POS" ]; then
    LAST_POS=0
fi

# Extract new content since last scan
NEW_CONTENT=$(tail -c +$((LAST_POS + 1)) "$LOG_FILE" 2>/dev/null)

# Save current position
echo "$CURRENT_SIZE" > "$LAST_SCAN_FILE"

# Scan for errors (case-insensitive)
ERRORS=$(echo "$NEW_CONTENT" | grep -i -E "error|exception|traceback|critical|fatal" | tail -5)

if [ -n "$ERRORS" ]; then
    ERROR_COUNT=$(echo "$NEW_CONTENT" | grep -ic -E "error|exception|traceback|critical|fatal")
    # Escape for JSON
    ESCAPED_ERRORS=$(echo "$ERRORS" | head -3 | sed 's/"/\\"/g' | tr '\n' ' ')
    curl -s -X POST http://localhost:8200/execute/v2 \
        -H "Content-Type: application/json" \
        -d "{\"task\":\"Post to #the-orb: Log scanner found $ERROR_COUNT errors in orb-backend.log in the last 30 minutes. Sample: $ESCAPED_ERRORS\"}" \
        >> "$SCAN_LOG" 2>&1
fi
```

Key design decisions:
- Tracks byte offset to only scan NEW log entries since last run
- Handles log rotation (file shrinks = reset offset)
- Matches: error, exception, traceback, critical, fatal (case-insensitive)
- Sends at most 3 sample error lines to avoid giant Slack messages
- Exits cleanly if log file doesn't exist

Create `/Users/claw2501/.openclaw/workspace/tools/trello-poll.sh`:

The script needs to detect NEW cards on the Trello Command Center board. Strategy:
1. Use the Trello API via curl (check `~/.claude/mcp.json` or environment for Trello API key/token — the user has a Trello MCP or API keys)
2. If no direct Trello API access, route through Smithers:

```bash
#!/bin/bash
# Trello Command Center poll — every 15 minutes
# Surfaces new cards to #the-orb via Smithers
export PATH=/opt/homebrew/bin:$PATH

STATE_FILE="/tmp/trello-poll-last-cards"
POLL_LOG="/Users/claw2501/.openclaw/workspace/logs/trello-poll.log"

# Use Smithers to check Trello (Smithers has MCP access to Trello)
RESULT=$(curl -s --max-time 30 -X POST http://localhost:8200/execute/v2 \
    -H "Content-Type: application/json" \
    -d '{"task":"Check Trello Command Center board for cards created or moved in the last 15 minutes. Return ONLY a JSON array of card names and list names, or empty array if none. Format: [{\"name\":\"card name\",\"list\":\"list name\"}]. No commentary."}')

# If Smithers returned card data and it differs from last state
if [ -n "$RESULT" ] && [ "$RESULT" != "[]" ]; then
    CURRENT_HASH=$(echo "$RESULT" | md5 -q 2>/dev/null || echo "$RESULT" | md5sum | cut -d' ' -f1)
    LAST_HASH=""
    if [ -f "$STATE_FILE" ]; then
        LAST_HASH=$(cat "$STATE_FILE")
    fi

    if [ "$CURRENT_HASH" != "$LAST_HASH" ]; then
        echo "$CURRENT_HASH" > "$STATE_FILE"
        # New cards detected — notify
        curl -s -X POST http://localhost:8200/execute/v2 \
            -H "Content-Type: application/json" \
            -d "{\"task\":\"Post to #the-orb: New Trello Command Center activity detected. Cards: $RESULT\"}" \
            >> "$POLL_LOG" 2>&1
    fi
fi

echo "$(date): poll complete" >> "$POLL_LOG"
```

Key design: Uses Smithers as the Trello bridge (Smithers has MCP access). Hashes results to detect changes. Only notifies when new/changed cards appear.

Make executable: `chmod +x`
