    ```python
    # ============================================================
    # Phase 15: JARVIS Agentic Build Loop
    # ============================================================

    SANDBOX_ROOT = Path("/Users/claw2501/esp32-jarvis/r1-frontend").resolve()

    ALLOWED_COMMANDS: frozenset = frozenset({
        "git status",
        "git diff",
    })
    # Note: r1-frontend is a plain static site (no package.json).
    # npm run build is NOT in the allowlist — no build step exists.


    def _assert_safe_path(raw_path: str) -> Path:
        """Raise ValueError if raw_path resolves outside SANDBOX_ROOT."""
        candidate = (SANDBOX_ROOT / raw_path).resolve()
        if not candidate.is_relative_to(SANDBOX_ROOT):
            raise ValueError(
                f"Path traversal blocked: {raw_path!r} resolves outside r1-frontend/"
            )
        return candidate


    async def _tool_read_file(path: str) -> str:
        """Read a file from r1-frontend/. Returns contents or error string."""
        try:
            safe = _assert_safe_path(path)
            return safe.read_text(encoding="utf-8")
        except ValueError as e:
            return f"ERROR: {e}"
        except FileNotFoundError:
            return f"ERROR: File not found: {path}"
        except Exception as e:
            return f"ERROR: {e}"


    async def _tool_write_file(path: str, content: str) -> str:
        """Write content to a file in r1-frontend/. Returns 'OK' or error string."""
        try:
            safe = _assert_safe_path(path)
            safe.parent.mkdir(parents=True, exist_ok=True)
            safe.write_text(content, encoding="utf-8")
            return "OK"
        except ValueError as e:
            return f"ERROR: {e}"
        except Exception as e:
            return f"ERROR: {e}"


    async def _tool_exec_shell(command: str) -> str:
        """Execute a shell command from ALLOWED_COMMANDS. Returns stdout+stderr or error."""
        if command not in ALLOWED_COMMANDS:
            return (
                f"ERROR: Command not allowed: {command!r}. "
                f"Allowed commands: {sorted(ALLOWED_COMMANDS)}"
            )
        try:
            loop = asyncio.get_event_loop()
            proc = await loop.run_in_executor(
                None,
                lambda: subprocess.run(
                    command,
                    shell=True,
                    capture_output=True,
                    text=True,
                    timeout=30,
                    cwd=str(SANDBOX_ROOT),
                )
            )
            return (proc.stdout + proc.stderr).strip() or "(no output)"
        except subprocess.TimeoutExpired:
            return "ERROR: Command timed out after 30 seconds"
        except Exception as e:
            return f"ERROR: {e}"


    async def _tool_reload_frontend() -> str:
        """Reload the R1 browser via ADB am start. Returns status string."""
        # Use am start re-navigation (reliable fallback; CDP requires remote-debug flag on R1).
        # Re-detect serial on each call to handle R1 reconnects.
        serial = get_r1_serial()
        if not serial:
            return "ERROR: R1 device not found via ADB"
        try:
            loop = asyncio.get_event_loop()
            proc = await loop.run_in_executor(
                None,
                lambda: subprocess.run(
                    [
                        "adb", "-s", serial, "shell",
                        "am", "start", "-a", "android.intent.action.VIEW",
                        "-d", "http://localhost:8888",
                        "com.android.chrome",
                    ],
                    capture_output=True,
                    text=True,
                    timeout=10,
                )
            )
            if proc.returncode != 0:
                return f"ERROR: ADB reload failed: {proc.stderr.strip()}"
            return "Frontend reloaded"
        except Exception as e:
            return f"ERROR: {e}"


    def _build_tool_definitions() -> list:
        return [
            {
                "name": "read_file",
                "description": (
                    "Read the contents of a file in the r1-frontend/ directory. "
                    "Pass a relative path like 'index.html' or 'js/thinking-mesh.js'. "

            {

                    "Only r1-frontend/ paths are writable. "
                    "Returns 'OK' or an error message."

                    "properties": {
                        "path": {
                            "type": "string",
                            "description": "Relative path from r1-frontend/ root"
                        },
                        "content": {
                            "type": "string",
                            "description": "Full file content to write (replaces entire file)"
                        }
                    },
                    "required": ["path", "content"]
                },
            },
            {
                "name": "exec_shell",
                "description": (

                    "'git status' or 'git diff'. "
                    "ANY other command is refused. Returns stdout+stderr or error."
                ),
                "input_schema": {

                        }
                    },
                    "required": ["command"]
                },
            },

                "name": "reload_frontend",
                "description": (
                    "Reload the R1 device browser to apply code changes. "
                    "Call this after writing a file to make the change visible. "
                    "Returns 'Frontend reloaded' on success or an error message."
                ),
                "input_schema": {
                    "type": "object",
                    "properties": {},
                    "required": []
                },
            },
        ]


    async def _dispatch_tool(name: str, inputs: dict) -> str:
        """Dispatch a tool call by name. Always returns a string (never raises)."""
        try:
            if name == "read_file":
                return await _tool_read_file(inputs.get("path", ""))
            elif name == "write_file":
                return await _tool_write_file(inputs.get("path", ""), inputs.get("content", ""))
            elif name == "exec_shell":
                return await _tool_exec_shell(inputs.get("command", ""))
            elif name == "reload_frontend":
                return await _tool_reload_frontend()
            else:
                return f"ERROR: Unknown tool: {name!r}"
        except Exception as e:
            return f"ERROR: Tool dispatch failed: {e}"


    async def run_build_loop(
        query: str,
        session_id: str,
        websocket,
        max_iterations: int = 5,
    ) -> str:
        """Bounded Claude tool-use loop for build-intent voice requests.

        AGEN-05: Fires TTS ACK before first API call. Caps at max_iterations.
        """
        # Fire voice ACK immediately — silence during 10-40s loop is not acceptable.
        await send_tts_morphing(websocket, "Working on it.", "jarvis")

        # Find the Anthropic client — it is module-level, search for it.
        # The variable is used throughout orb_bridge.py for LLM calls.
        # (Executor: grep for `anthropic.Anthropic\|Anthropic(` to find exact var name,
        #  then use it below instead of `anthropic_client` if named differently.)
        system_prompt = (
            "You are JARVIS, a voice AI assistant. You have tools to read and modify "
            "files in the r1-frontend/ directory of an R1 device running a Three.js "
            "orb visualization. When the user asks you to change something visual "
            "(e.g., 'make the orb pulse faster'), read the relevant JS file first, "
            "make the targeted change, write it back, then call reload_frontend. "
            "Keep your final confirmation short — it will be spoken aloud. "
            "IMPORTANT: You can only read and write files inside r1-frontend/. "
            "You cannot access any other directory. "
            "If you cannot complete the request after reading files, say so clearly."
        )

        messages = [{"role": "user", "content": query}]
        tools = _build_tool_definitions()
        last_tool_name = "(none)"
        iterations = 0

        while True:
            # AGEN-05: Cap check BEFORE API call — prevents runaway spend
            if iterations >= max_iterations:
                summary = (
                    f"I reached my limit of {max_iterations} steps without finishing. "
                    f"The last tool I tried was {last_tool_name}. "
                    "Please try a more specific request."
                )
                await send_tts_morphing(websocket, summary, "jarvis")
                print(f"  [BUILD] Loop cap hit after {iterations} iterations. Last tool: {last_tool_name}")
                return summary

            iterations += 1
            print(f"  [BUILD] Iteration {iterations}/{max_iterations}, last_tool={last_tool_name}")

            try:
                response = anthropic_client.messages.create(
                    model="claude-sonnet-4-6",
                    max_tokens=4096,
                    system=system_prompt,
                    tools=tools,
                    messages=messages,
                )
            except Exception as e:
                error_msg = f"I ran into an error while working on your request: {e}"
                await send_tts_morphing(websocket, error_msg, "jarvis")
                return error_msg

            messages.append({"role": "assistant", "content": response.content})

            if response.stop_reason == "end_turn":
                final_text = next(
                    (b.text for b in response.content if hasattr(b, "text")),
                    "Done.",
                )
                await send_tts_morphing(websocket, final_text, "jarvis")
                print(f"  [BUILD] Completed in {iterations} iteration(s)")
                return final_text

            if response.stop_reason != "tool_use":
                # Unexpected stop (max_tokens, refusal, etc.)
                msg = f"I stopped unexpectedly ({response.stop_reason}). Please try again."
                await send_tts_morphing(websocket, msg, "jarvis")
                return msg

            # Execute all tool_use blocks in this response
            tool_results = []
            for block in response.content:
                if not hasattr(block, "type") or block.type != "tool_use":
                    continue
                last_tool_name = block.name
                result_content = await _dispatch_tool(block.name, block.input)
                print(f"  [BUILD] Tool {block.name}({block.input}) -> {result_content[:120]!r}")
                tool_results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": result_content,
                })

            messages.append({"role": "user", "content": tool_results})
    ```
